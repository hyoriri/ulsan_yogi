function haversine(lat1, lon1, lat2, lon2) {
  const R = 6371;
  const dLat = (lat2 - lat1) * Math.PI / 180;
  const dLon = (lon2 - lon1) * Math.PI / 180;
  const a = Math.sin(dLat/2)**2 + Math.cos(lat1*Math.PI/180)*Math.cos(lat2*Math.PI/180)*Math.sin(dLon/2)**2;
  return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
}

function formatDist(km) {
  const t = I18N[currentLang];
  if (km < 1) return `${t.distance_prefix} ${Math.round(km*1000)}m ${t.distance_suffix}`.trim();
  return `${t.distance_prefix} ${km.toFixed(1)}km ${t.distance_suffix}`.trim();
}

function sortByDistance() {
  if (userLat === null) return;
  sortedSpots = [...SPOTS].map(s => ({
    ...s,
    dist: haversine(userLat, userLng, s.lat, s.lng)
  })).sort((a, b) => a.dist - b.dist);
  renderSpots();
}

// ===== GPS =====
function getMyLocation() {
  const status = document.getElementById('location-status');
  status.classList.remove('hidden');
  status.textContent = I18N[currentLang].gps_loading;

  if (!navigator.geolocation) {
    status.textContent = I18N[currentLang].gps_error;
    return;
  }
  navigator.geolocation.getCurrentPosition(
    (pos) => {
      userLat = pos.coords.latitude;
      userLng = pos.coords.longitude;
      status.textContent = I18N[currentLang].gps_found;
      sortByDistance();
      setTimeout(() => status.classList.add('hidden'), 2000);
    },
    () => {
      status.textContent = I18N[currentLang].gps_error;
    }
  );
}

  if (key) {

    sortByDistance();
    setTimeout(() => status.classList.add('hidden'), 2000);

    // Default to Ulsan city center
    userLat = 35.5397;
    userLng = 129.3114;
    status.textContent = I18N[currentLang].gps_found + ' (울산 시청 기준)';
    sortByDistance();
    setTimeout(() => status.classList.add('hidden'), 3000);
  }
}

function toggleDetail(id) {
  const panel = document.getElementById(`detail-${id}`);
  const btn = document.getElementById(`toggle-${id}`);
  const t = I18N[currentLang];
  const isOpen = panel.classList.contains('open');

  panel.classList.toggle('open');
  const icon = btn.querySelector('i');
  icon.style.transform = isOpen ? '' : 'rotate(180deg)';
  btn.querySelector('span').textContent = isOpen ? t.detail_btn : t.close_btn;
}

// ===== LANGUAGE =====
function setLanguage(lang) {
  currentLang = lang;
  const t = I18N[lang];

  // Update lang buttons
  document.querySelectorAll('.lang-btn').forEach(b => {
    b.classList.toggle('active', b.dataset.lang === lang);
  });

  // Update static i18n elements
  document.querySelectorAll('[data-i18n]').forEach(el => {
    const key = el.dataset.i18n;
    if (t[key]) el.textContent = t[key];
  });
  document.querySelectorAll('[data-i18n-placeholder]').forEach(el => {
    const key = el.dataset.i18nPlaceholder;
    if (t[key]) el.placeholder = t[key];
  });

  // Update category buttons
  document.querySelectorAll('.cat-btn').forEach(b => {
    const key = b.dataset.i18n;
    if (key && t[key]) b.textContent = t[key];
  });

  // Update html lang
  document.documentElement.lang = lang === 'ko' ? 'ko' : lang === 'ja' ? 'ja' : lang === 'zh' ? 'zh' : 'en';

  renderSpots();
}

// ===== CATEGORY FILTER =====
function setCategory(cat) {
  currentCat = cat;
  document.querySelectorAll('.cat-btn').forEach(b => {
    if (b.dataset.cat === cat) {
      b.className = 'cat-btn active bg-[#3e4f5f] text-white px-4 py-2 rounded-full text-sm transition';
    } else {
      b.className = 'cat-btn bg-white text-[#5a6a70] px-4 py-2 rounded-full text-sm hover:bg-[#3e4f5f] hover:text-white transition';
    }
  });
  renderSpots();
}
